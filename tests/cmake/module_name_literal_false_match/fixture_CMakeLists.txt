cmake_minimum_required(VERSION 3.31)
project(module_name_false_match_fixture LANGUAGES CXX)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(gentest_BUILD_TESTING OFF CACHE BOOL "" FORCE)
set(GENTEST_BUILD_CODEGEN OFF CACHE BOOL "" FORCE)
add_subdirectory("${GENTEST_SOURCE_DIR}" gentest-build EXCLUDE_FROM_ALL)
add_executable(module_name_false_match_tests)
target_compile_features(module_name_false_match_tests PRIVATE cxx_std_20)
target_include_directories(module_name_false_match_tests PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/inc")
target_link_libraries(module_name_false_match_tests PRIVATE gentest::gentest gentest::gentest_main)
target_sources(module_name_false_match_tests PRIVATE FILE_SET module_cases TYPE CXX_MODULES FILES
  "${CMAKE_CURRENT_SOURCE_DIR}/conditional_false_match.ixx")
gentest_attach_codegen(module_name_false_match_tests OUTPUT_DIR "${CMAKE_CURRENT_BINARY_DIR}/gentest_codegen")
